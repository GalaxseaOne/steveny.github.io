🔫 Gobuster

Gobuster — Installation & Commandes
# Installation
sudo apt install gobuster
go install github.com/OJ/gobuster/v3@latest

# Découverte de répertoires (mode dir)
gobuster dir -u http://target.com -w /usr/share/wordlists/dirb/common.txt

# Avec extensions
gobuster dir -u http://target.com -w common.txt -x php,html,txt,bak,sql

# Avec threads (plus rapide)
gobuster dir -u http://target.com -w common.txt -t 50

# Avec cookie de session
gobuster dir -u http://target.com -w common.txt -c "PHPSESSID=abc123"

# Ignorer certains codes de réponse
gobuster dir -u http://target.com -w common.txt --exclude-length 0

# Énumération de sous-domaines (mode dns)
gobuster dns -d target.com -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt

# Mode vhost (virtual hosts)
gobuster vhost -u http://target.com -w subdomains.txt --append-domain

# Output dans fichier
gobuster dir -u http://target.com -w common.txt -o results.txt

🚀 Feroxbuster

Feroxbuster — Récursif et rapide
# Installation
sudo apt install feroxbuster
curl -sL https://raw.githubusercontent.com/epi052/feroxbuster/main/install-nix.sh | bash

# Scan basique
feroxbuster -u http://target.com

# Avec wordlist personnalisée
feroxbuster -u http://target.com -w /usr/share/seclists/Discovery/Web-Content/raft-large-directories.txt

# Avec extensions (récursif par défaut)
feroxbuster -u http://target.com -x php,html,txt,js

# Profondeur maximale
feroxbuster -u http://target.com -d 3

# Filtrer par taille de réponse
feroxbuster -u http://target.com --filter-size 0

# Threads
feroxbuster -u http://target.com -t 100

# Mode silencieux avec sortie JSON
feroxbuster -u http://target.com --json -o results.json -q

📂 Dirb

Dirb
# Scan basique (wordlist intégrée)
dirb http://target.com

# Wordlist personnalisée
dirb http://target.com /usr/share/wordlists/dirb/big.txt

# Avec extensions
dirb http://target.com -X .php,.html,.txt

# Avec cookies
dirb http://target.com -c "cookie=value"

# Output fichier
dirb http://target.com -o output.txt

# Ignorer les codes
dirb http://target.com -N 404

⚡ ffuf — Fuzzing rapide

ffuf
# Installation
sudo apt install ffuf
go install github.com/ffuf/ffuf/v2@latest

# Répertoires web
ffuf -u http://target.com/FUZZ -w wordlist.txt

# Filtrer par code HTTP
ffuf -u http://target.com/FUZZ -w wordlist.txt -mc 200,301,302

# Filtrer par taille
ffuf -u http://target.com/FUZZ -w wordlist.txt -fs 1234

# Fuzzing de paramètres GET
ffuf -u "http://target.com/page?FUZZ=test" -w params.txt

# Fuzzing POST
ffuf -u http://target.com/login -X POST -d "username=FUZZ&password=test" -w users.txt

# Sous-domaines (vhosts)
ffuf -u http://target.com -H "Host: FUZZ.target.com" -w subdomains.txt -fs 1234

# Extensions + répertoires combinés
ffuf -u http://target.com/FUZZ -w wordlist.txt -e .php,.html,.js,.bak
