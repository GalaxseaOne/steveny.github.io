📦 Installation

Installation
# Kali / Ubuntu (version basique)
sudo apt install john

# Version Jumbo (plus de formats, recommandée)
sudo apt install john
# OU depuis les sources pour la version complète :
git clone https://github.com/openwall/john.git
cd john/src && ./configure && make -s clean && make -sj4
cd ../run && ./john --list=formats

# Vérifier les formats supportés
john --list=formats
john --list=formats | grep -i ntlm
john --list=formats | grep -i sha

📋 Cheatsheet — John the Ripper

Cracking de base

Commandes essentielles
# Auto-détection du format et dictionnaire
john --wordlist=/usr/share/wordlists/rockyou.txt hashes.txt

# Spécifier le format manuellement
john --format=md5crypt --wordlist=rockyou.txt hashes.txt
john --format=NT --wordlist=rockyou.txt hashes.txt
john --format=bcrypt --wordlist=rockyou.txt hashes.txt

# Afficher les mots de passe crackés
john --show hashes.txt
john --show --format=NT hashes.txt

# Mode brute force pur (lent)
john --incremental hashes.txt

# Mode single (utilise les infos du fichier)
john --single hashes.txt

# Reprendre un cracking interrompu
john --restore

Outils de conversion (john2format)

*2john — Convertir pour John
# Fichier /etc/shadow Linux
unshadow /etc/passwd /etc/shadow > shadow_combined.txt
john --wordlist=rockyou.txt shadow_combined.txt

# Clé SSH privée chiffrée
ssh2john id_rsa > id_rsa_hash.txt
john --wordlist=rockyou.txt id_rsa_hash.txt

# Archive ZIP
zip2john archive.zip > zip_hash.txt
john --wordlist=rockyou.txt zip_hash.txt

# Archive RAR
rar2john archive.rar > rar_hash.txt
john --wordlist=rockyou.txt rar_hash.txt

# PDF protégé
pdf2john document.pdf > pdf_hash.txt
john --wordlist=rockyou.txt pdf_hash.txt

# KeePass
keepass2john database.kdbx > keepass_hash.txt
john --wordlist=rockyou.txt keepass_hash.txt

# Office 2007+
office2john document.docx > office_hash.txt
john --wordlist=rockyou.txt office_hash.txt

# Hash NTLMv2 capturé avec Responder
john --format=netntlmv2 --wordlist=rockyou.txt ntlmv2_hashes.txt

Règles de mutation

Règles (Rules)
# Avec règles par défaut
john --wordlist=rockyou.txt --rules hashes.txt

# Règles spécifiques
john --wordlist=rockyou.txt --rules=Jumbo hashes.txt
john --wordlist=rockyou.txt --rules=KoreLogic hashes.txt

# Les règles appliquent des transformations :
# password → Password, P@ssword, password1, PASSWORD, p4ssw0rd, etc.

# Créer une wordlist mutée
john --wordlist=rockyou.txt --rules --stdout > mutated_list.txt

⚡ Hashcat — Alternative GPU

Hashcat (GPU — beaucoup plus rapide)
# Installation
sudo apt install hashcat

# Attaque par dictionnaire (-a 0)
hashcat -m 0 -a 0 hash.txt rockyou.txt
# -m 0 = MD5   -m 100 = SHA1   -m 1000 = NTLM
# -m 1800 = sha512crypt   -m 3200 = bcrypt
# -m 22000 = WPA-PBKDF2   -m 5600 = NTLMv2

# Brute force masque (-a 3)
hashcat -m 0 -a 3 hash.txt "?l?l?l?l?l?l?l?l"
# ?l=minuscule ?u=majuscule ?d=chiffre ?s=spécial ?a=tous

# Règles
hashcat -m 1000 -a 0 hash.txt rockyou.txt -r /usr/share/hashcat/rules/best64.rule

# Afficher résultats
hashcat -m 0 --show hash.txt

# Identifier un hash
hashcat --identify 5f4dcc3b5aa765d61d8327deb882cf99

# Modes courants (-m) :
0     MD5
100   SHA1
1000  NTLM
1800  sha512crypt ($6$)
3200  bcrypt ($2*)
5600  NTLMv2
13100 Kerberoast (TGS-REP)
22000 WPA-PBKDF2-PMKID+EAPOL
💡
Identifier un hash inconnu : Utilisez hashes.com ou hashid hash.txt pour identifier le format avant de lancer John ou Hashcat.
